\Procedure{Primer}{}

    \State Inicializiraj $b$ na 1
    \For{$i \gets 1$ to $n$}
        \State $c \gets a + b$
        \State $a \gets b$
        \State $b \gets c$
    \EndFor

\EndProcedure
